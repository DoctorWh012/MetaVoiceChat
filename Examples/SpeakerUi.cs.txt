using Assets.Metater;
using UnityEngine;
using UnityEngine.UI;

public class SpeakerUi : MetaMbs<SpeakerUi>
{
    [Header("Colors")]
    public Color activeColor;
    public Color inactiveColor;

    [Header("References")]
    public Image image;
    public Sprite[] speakerSprites;

    public void SetIndex(int i, bool isActive = true)
    {
        if (i < 0 || i >= speakerSprites.Length)
        {
            image.enabled = false;
            return;
        }

        if (isActive)
        {
            image.color = activeColor;
        }
        else
        {
            image.color = inactiveColor;
        }

        image.enabled = true;
        image.sprite = speakerSprites[i];
    }
}
