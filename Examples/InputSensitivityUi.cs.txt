using Assets.Metater;
using R3;
using UnityEngine.UI;

public class InputSensitivityUi : MetaMbs<InputSensitivityUi>
{
    public Slider activitySlider;

    public SerializableReactiveProperty<float> activity = new(0f);

    private void Awake()
    {
        activity.Subscribe(activity =>
        {
            activitySlider.SetValueWithoutNotify(activity);
        }).AddTo(this);
    }
}
